mysql指令

在mysql5.7中用户表password已经变成了authentication_string

修改密码指令改成

update mysql.user set authentication_string=password('*******') where user='*******';  update mysql.user set authentication_string=password('*******') where user='*******';  #修改密码成功

mysql创建用户与刷新权限(5.6)

-- 插入数据表
insert into mysql.user(host,user,password) values("localhost","hzj",password("upyun123"));
-- 刷新权限表
flush privileges;

mysql5.6更新用户密码

-- 记住要加password
update user set password=password("upyun123") where user="hzj" and host="localhost";

用户创建授权

# 为本地的hzj用户开放权限
mysql> grant all on *.* to 'hzj'@'localhost';
# 为远程主机的hzj用户开放权限
mysql> grant all on *.* to 'hzj'@'%';
# 为指定IP主机登录的hzj用户开放权限
mysql> grant all on *.* to 'hzj'@'192.168.20.1';

修改数据库字符集(5.6)

默认情况下数据库的字符集是latin1,我们需要修改成utf8mb4;

ALTER DATABASE ALTER DATABASE `switch_instance`
DEFAULT CHARACTER SET utf8mb4
DEFAULT COLLATE utf8mb4_general_ci;

创建的时候，直接初始化数据库为utf8mb4

CREATE DATABASE select_test character CREATE DATABASE select_test character set utf8mb4;

使用shell操作数据库

#!/bin/bash
# 连接数据库
mysql_instance="mysql -uroot -p123456 -A -N --default-character-set=utf8mb4"
# 操作数据库
sql_get_data="select * from mysql.user"
# 执行命令返回结果
result="$($mysql_instance -e "$sql_get_data")"

# 按照格式输出
echo -e "$result"

看看别人的写法

#!/bin/bash
HOSTNAME="192.168.111.84" #数据库信息
PORT="3306"
USERNAME="root"
PASSWORD=""
DBNAME="test_db_test" #数据库名称
TABLENAME="test_table_test" #数据库中表的名称
#创建数据库
create_db_sql="create database ${DBNAME} IF NOT EXISTS ${DBNAME}"
mysql -h${HOSTNAME} -P${PORT} -u${USERNAME} -p${PASSWORD} -e "${create_db_sql}"
#创建表
create_table_sql="create table ${TABLE} IF NOT EXISTS ${TABLENAME} ( name varchar(20), id int(11) default 0 )"
mysql -h${HOSTNAME} -P${PORT} -u${USERNAME} -p${PASSWORD} ${DBNAME} -e "${create_table_sql}"
#插入数据
insert_sql="insert into ${TABLENAME} values('billchen',2)"
mysql -h${HOSTNAME} -P${PORT} -u${USERNAME} -p${PASSWORD} ${DBNAME} -e "${insert_sql}"
#查询
select_sql="select * from ${TABLENAME}"
mysql -h${HOSTNAME} -P${PORT} -u${USERNAME} -p${PASSWORD} ${DBNAME} -e "${select_sql}"
#更新数据
update_sql="update ${TABLENAME} set id=3"
mysql -h${HOSTNAME} -P${PORT} -u${USERNAME} -p${PASSWORD} ${DBNAME} -e "${update_sql}"
mysql -h${HOSTNAME} -P${PORT} -u${USERNAME} -p${PASSWORD} ${DBNAME} -e "${select_sql}"
#删除数据
delete_sql="delete from ${TABLENAME}"
mysql -h${HOSTNAME} -P${PORT} -u${USERNAME} -p${PASSWORD} ${DBNAME} -e "${delete_sql}"
mysql -h${HOSTNAME} -P${PORT} -u${USERNAME} -p${PASSWORD} ${DBNAME} -e "${select_sql}"
