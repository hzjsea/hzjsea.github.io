golang并发

golang并发的简单实例

package main

import (
    "fmt"
)

func main() {
    go helloworld()
    //time.Sleep(2*time.Second)
}

func helloworld() {
    fmt.Println("helloworld")
}

如上如果我们把延时注释掉执行程序的话，是不会有任何内容输出的，但是加上延时后，会输出”helloworld”
这是因为go会使用线程池来调用一个子程序来完成helloworld 也就是go后面程序。而延时主要是保证主程序还在，这样子程序才能正常运行

双向channel

  • Channel 是 goroutine 沟通的桥梁，大都是阻塞同步的
  • 它是通过 make 创建，close 关闭
  • Channel 是引用类型
  • 可以使用 for range 来迭代，不断操作 channel
  • 可以设置单向 或 双向通道
  • 可以设置缓存大小，在未被填满前不会发生阻塞，即它是异步的
  • 那么针对上溯代码我们不使用休眠，而使用 Channel 来实现我们想要的效果：
    channel的意思用白话可以这么理解：主线程告诉大家你开goroutine可以，但是我在我的主线程开了一个管道，你做完了你要做的事情之后，往管道里面塞个东西告诉我你已经完成了
package main

import (
    "fmt"
)

func main() {
    //声明创建一个通道，存储类型为bool型 这个通道是双向通道，可以存取
    //注意channel 是一个引用类型 和 slice map 类似
    c := make(chan bool)
    //启用一个goroutine,使用的是匿名方法方式
    go func() {
        fmt.Println("Go Go Go!!!")
        c <- true  //向 channel 中存入一个值
    }()

    // <- c // 释放值 表示线程结束
    fmt.Println(<- c)
    /**
      主线程启动了一个匿名子线程后就执行到了：<-c , 到达这里主线程就被阻塞了。只有当子线程向通道放入值后主线程阻塞才会被释放
      其实这个就是完成了消息的发送
    */
}
package main

import (
    "fmt"
)

func main() {
    //声明创建一个通道，存储类型为bool型
    //注意channel 是一个引用类型 和 slice map 类似
    c := make(chan bool)
    //启用一个goroutine,使用的是匿名方法方式
    go func() {
        fmt.Println("Go Go Go!!!")
        c <- true  //向 channel 中存入一个值
        close(c)
    }()
     for value:= range c {
         fmt.Println(value)
    }
}

如上使用如果我们使用 <- c 表示子程序也释放结束了，但是我们可以使用 close(c)来关闭子程序，保证主程序还在运行，并使用rang获取内容

channel缓存

  • make(chan bool, 1) 表示带有一个缓存大小的缓存channel
  • make(chan bool) 或 make(chan bool, 0) 表示一个无缓存的channel
  • 无缓存channel是阻塞的即同步的，而有缓存channel是异步的。

如下

package main

import "time"

func main() {
    c := make(chan bool)
    c <- true
    time.Sleep(2*time.Second)
    <- c
}

这样执行是错误的，因为他是无缓存的channel 也就是说当执行c <- true的时候，一定要有别的线程 <- true 接手了这个参数 那么 c <- true 才会执行下去
而上面整段都在main程序里面，也就是都在主程序里面，这样就无法执行下去，会报错

package main

import "time"

func main() {
    c := make(chan bool,1)
    go func(){
        c <- true
        time.Sleep(2*time.Second)
    }()
    <- c
}

这样就可以执行，因为main接管了子程序的 <-c

有缓存的channel，如下

package main

import "time"

func main() {
    c := make(chan bool,1)
    c <- true
    time.Sleep(2*time.Second)
    <- c

}

可以正常运行

多线程抢占执行

在多线程环境下，多个线程并发抢占会使得打印不是按照顺序来，那么我们如何确保子线程全部结束完之后主线程再停止呢？主要有两种方式

一使用阻塞channel

package main

import (
    "fmt"
    "runtime"
)

func main() {
    // 输出当前系统的CPU数量
    fmt.Println(runtime.NumCPU())
    // 设置当前程序执行使用的CPU数量
    runtime.GOMAXPROCS(runtime.NumCPU()/2)

    // 创建一个channel
    c1 := make(chan  bool)

    for i:=0;i<10;i++{
        go goRun(c1,i)
    }
    for i := 0; i < 10; i++ {
        <- c1
    }
}


func goRun(c chan bool,index int ) {
    sum := 0
    for i:=0;i<100000000;i++{
        sum+=i
    }
    fmt.Println("线程序号:", index, sum)
    c <- true
}

从打印结果可以看出，多线程环境下运行代码打印和顺序没有关系，由 CPU 调度自己决定，多运行几次打印结果一定不会一样，就是这个道理。

二使用同步机制

package main

import (
    "fmt"
    "runtime"
    "sync"
)

func main() {
    fmt.Println("当前系统核数：", runtime.NumCPU())
    runtime.GOMAXPROCS(runtime.NumCPU()) //设置当前程序执行使用的并发数
    /**
      waitGroup即任务组，它的最要作用就是用来添加需要工作的任务，没完成一次任务就标记一次Done，这样任务组的待完成量会随之减1
      那么主线程就是来判断任务组内是否还有未完成任务，当没有未完成当任务之后主线程就可以结束运行，从而实现了与阻塞队列类似的同步功能
      这里创建了一个空的waitGroup(任务组)
    */
    wg := sync.WaitGroup{}
    wg.Add(10)  //添加10个任务到任务组中
    //这里启动10个线程运行
    for i :=0; i < 10; i++ {
        go goRun(&wg, i)
    }
    wg.Wait()
}

/**
这里需要传入引用类型不能传入值拷贝，因为在子线程中是需要执行Done操作，类似与我们修改结构体中的int变量主词递减，如果是只拷贝的话是不会影响原类型内的数据
这样就会发生死循环导致死锁程序奔溃，报错异常为：fatal error: all goroutines are asleep - deadlock!
*/
func goRun(wg *sync.WaitGroup, index int) {
    a := 1
    //循环叠加1千万次并返回最终结果
    for i := 0; i < 10000000; i++ {
        a += i
    }
    fmt.Println("线程序号:", index, a)

    wg.Done()
}

使用Select处理多个channel

package main

import "fmt"

func main() {
    // 创建多个channel
    c1,c2 := make(chan int),make(chan  string)
    go func() {
        /**
          创建一个无限循环语句,使用select进行处理
          我们一般都是使用这种方式来处理不断的消息发送和处理
        */
        for {
            select {
            case v, ok := <- c1:
                if !ok {
                    break
                }
                fmt.Println("c1:", v)
            case v, ok := <- c2:
                if !ok {
                    break
                }
                fmt.Println("c2:", v)
            }
        }
    }()

    c1 <- 1
    c2 <- "liang"
    c1 <- 2
    c2 <- "xuli"

    //关闭channel
    close(c1)
    close(c2)
}
