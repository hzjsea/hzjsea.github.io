vec的官方库

来看看vec的官方库

官方库 https://doc.rust-lang.org/std/vec/index.html
集合的官方库 https://doc.rust-lang.org/std/vec/struct.Vec.html

基本类型

fn main(){
    let mut vec = Vec::new();
    vec.push(1);
    vec.push(2);

    assert_eq!(vec.len(), 2);
    assert_eq!(vec[0], 1);

    assert_eq!(vec.pop(), Some(2));
    assert_eq!(vec.len(), 1);

    vec[0] = 7;
    assert_eq!(vec[0], 7);

    vec.extend([1, 2, 3].iter().copied());

    for x in &vec {
        println!("{}", x);
    }
    assert_eq!(vec, [7, 1, 2, 3]);
}

append集合合并

pub fn append(&mut self, other: &mut Vec<T>)

接收一个可变集合的自身引用，以及另外一个可变集合的引用

fn main(){
    let mut v1=vec![1,2,3,45];
    let mut v2= vec![1,2,3,4,5];
    v1.append(&mut v2);

    println!("{:?}",v1)
}

拖动集合中的元素drain

fn main(){
    let mut  v = vec![1,2,3,4];
    let v2: Vec<_>  = v.drain(1..).collect();

    // 大致的意思就是 集合v中的元素拖动到v2中去
    

    assert_eq!(v,[1]);
    // assert_eq!(v,&[1]);
    assert_eq!(v2,[2,3,4]);
    // assert_eq!(v2,&[2,3,4]);

    // 全部拖动
    let mut v3 = vec![1,2,3,4];
    v3.drain(..);

    assert_eq!(v3,[]);
    // assert_eq!(v3,&[]);

}

清空集合clear

pub fn clear(&mut self)

接收一个可变参数自身

fn main(){
    let mut  v = vec![1,2,3,4];
    // 判断是否为空集合
    v.drain(..);
    println!("{}",v.is_empty());

    // 清空集合  被清空的集合必须是可变集合
    let mut v3 = vec![1,2,34];
    v3.clear();
    println!("{}",v3.is_empty());
}

截取集合成为一个新的集合 split_off

pub fn split_off(&mut self, at: usize) -> Vec<T>

获取一个自身可变参数和一个截取序号，返回一个集合

当at > vec.len()的时候就会出现panic报错

fn main(){
    let mut v = vec![1,2,3,4];
    let _new_v:Vec<_> = v.split_off(2);
    assert_eq!(_new_v,[3,4]);
}

截断或者补充 resize_with

pub fn resize_with<F>(&mut self, new_len: usize, f: F)

接收一个可变自身，一个新的长度，以及填充物
当new_len > vec.len()的时候，会填充后面的f内容
当new_len < vec.len()的时候，会进行截断形成新的集合

fn main(){
    let mut v = vec![1,2,3,4];

    // 输出集合长度
    println!("{}",v.len());

    // 截断
    v.resize_with(2,Default::default);
    assert_eq!(v,[1,2]);
    println!("{:?}",v);

    // 填充
    v.resize_with(8, Default::default);
    assert_eq!(v,[1,2,0,0,0,0,0,0]);
    println!("{:?}",v);

}

截断或者补充 resize

pub fn resize(&mut self, new_len: usize, value: T)

这个和resize_with差不多

fn main(){
    let mut v = vec![1,2,3,4];

    // 输出集合长度
    println!("{}",v.len());

    // 截断
    v.resize(2,0);
    assert_eq!(v,[1,2]);

    println!("{:?}",v);
    // 填充
    v.resize(8, 0);
    assert_eq!(v,[1,2,0,0,0,0,0,0]);
    println!("{:?}",v);  
}
