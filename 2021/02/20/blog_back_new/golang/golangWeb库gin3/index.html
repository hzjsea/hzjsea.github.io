// initRouter/initRouter.go

package initRouter

import (
	"github.com/gin-gonic/gin"
	"net/http"
)

func SetupRouter() *gin.Engine{
	router := gin.Default()
	router.GET("/", func(c *gin.Context) {
		c.String(200,"helloworld")
	})
	router.GET("/ping", func(c *gin.Context) {
		c.String(http.StatusOK,"pong")
	})
	return router
}

// main.go
package main

import (
	"ginRestfulDemo/initRouter"
)

func main() {
	// 路由
	router := initRouter.SetupRouter()
	router.Run()
}

// test/index_test.go
package test

import (
	"ginRestfulDemo/initRouter"
	"github.com/stretchr/testify/assert"
	"net/http"
	"net/http/httptest"
	"testing"
)

func TestIndexGetRouter(t *testing.T) {
	router := initRouter.SetupRouter()
	w := httptest.NewRecorder()
	req, _ := http.NewRequest(http.MethodGet, "/", nil)
	router.ServeHTTP(w, req)
	assert.Equal(t, http.StatusOK, w.Code)
	assert.Equal(t, "helloworld", w.Body.String())
}

其他的几种请求方法

package initRouter

import (
	"github.com/gin-gonic/gin"
	"net/http"
)

func SetupRouter() *gin.Engine{
	router := gin.Default()

	// Get
	router.GET("/", func(c *gin.Context) {
		c.String(200,"helloworld")
	})
	router.GET("/ping", func(c *gin.Context) {
		c.String(http.StatusOK,"pong")
	})

	// Post
	router.POST("/ping", func(c *gin.Context) {
		c.String(http.StatusOK,"pong")
	})

	// delete
	router.DELETE("/ping", func(c *gin.Context) {
		c.String(http.StatusOK,"pong")
	})

	// Put
	router.PUT("/ping", func(c *gin.Context) {
		c.String(http.StatusOK, "pong")
	})

	// 添加 Patch 请求路由
	router.PATCH("/ping", func(context *gin.Context) {
		context.String(http.StatusOK, "hello gin patch method")
	})

	// 添加 Head 请求路由
	router.HEAD("/ping", func(context *gin.Context) {
		context.String(http.StatusOK, "hello gin head method")
	})

	// 添加 Options 请求路由
	router.OPTIONS("/ping", func(context *gin.Context) {
		context.String(http.StatusOK, "pong")
	})

	return router
}

简化代码，因为每次请求之后的响应体是一样的，所以可以把响应的方法统一写成一个方法

package initRouter

import (
	"github.com/gin-gonic/gin"
	"net/http"
)


func SetupRouter() *gin.Engine{
	router := gin.Default()
	router.GET("/ping", OuputPong)
	router.POST("/ping",OuputPong)
	router.DELETE("/ping", OuputPong)
	router.PUT("/ping",OuputPong)
	router.PATCH("/ping",OuputPong)
	router.HEAD("/ping",OuputPong)
	router.OPTIONS("/ping", OuputPong)
	return router
}


func OuputPong(c *gin.Context){
	c.String(http.StatusOK,"pong")
}
